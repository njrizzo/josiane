CREATE TABLE curso
(
 
 codcurso serial PRIMARY KEY,
  nome character varying(100) NOT NULL,
 
 modulo character varying(100) NOT NULL,
 
 descricao character varying(100) NOT NULL,
  
cargahr float NOT NULL,
 
 areatema character varying(100) NOT NULL,
 
 competencia character varying(100) NOT NULL,
  
estado character varying(100) NOT NULL
);

CREATE TABLE turma(
codturma serial PRIMARY KEY,

nometurma VARCHAR (50) NOT NULL,
diasemana VARCHAR (50) NOT NULL,
datainicio date NOT NULL,
datafim date NOT NULL,
horainicio time NOT NULL,
horafim time NOT NULL,
codcurso integer NOT NULL references curso (codcurso)  ON DELETE CASCADE
);
 




CREATE TABLE inscricao(
codinscricao serial PRIMARY KEY,
datains date,
situacao VARCHAR (15) NOT NULL,
motivo VARCHAR (100),
justificativa VARCHAR (100),
codserv integer NOT NULL references servidor (codserv)  ON DELETE CASCADE,
codturma integer NOT NULL references turma (codturma)  ON DELETE CASCADE,
UNIQUE (codserv,codturma)
);




CREATE TABLE matricula(
codmatricula serial 	PRIMARY KEY,
datamat date,
situacao VARCHAR (15) NOT NULL,
codserv integer NOT NULL references servidor (codserv)  ON DELETE CASCADE,
codturma integer NOT NULL references turma (codturma)  ON DELETE CASCADE,
UNIQUE (codserv,codturma)
);



create table recuperacao(
codrec serial PRIMARY KEY,
utilizador  VARCHAR(50) NOT NULL,
confirmacao VARCHAR(40) NOT NULL,
data date
);


create table certificados(
codcert serial PRIMARY KEY,
chave VARCHAR(40) NOT NULL,
data date
);
